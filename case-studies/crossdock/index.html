Code Sample

Gmail Parser (Apps Script)

Google Apps Script · gmail_parser.gs 
function processCrossDockEmails() {
  const threads = GmailApp.search(
    'subject:"CrossDock Shipment" is:unread'
  );

  threads.forEach(thread => {
    const msg     = thread.getMessages()[0];
    const body    = msg.getPlainBody();
    const sender  = msg.getFrom();

    // Extract shipment fields via regex
    const data = {
      shipment_id: extract(body, /Shipment[#\s:]+(\w+)/i),
      carrier:     extract(body, /Carrier[:\s]+(.+)/i),
      eta:         extract(body, /ETA[:\s]+(.+)/i),
      qty:         extract(body, /Units[:\s]+(\d+)/i),
    };

    // Send acknowledgment reply
    thread.reply(buildAckTemplate(data));

    // Log to tracking sheet
    appendToSheet(data, sender, new Date());

    // Mark as read
    thread.markRead();
  });
}
